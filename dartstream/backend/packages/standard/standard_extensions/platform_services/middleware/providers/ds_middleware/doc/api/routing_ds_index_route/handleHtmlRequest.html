void handleHtmlRequest(HttpRequest request) {
  request.response
    ..statusCode = HttpStatus.ok
    ..headers.contentType = ContentType.html
    ..write('<html><body><h1>Welcome, To the main page!</h1></body></html>')
    ..close();
}
